namespace = color_picker

color_picker.2 = {
	type = country_event
	hidden = yes

	trigger = {
		OR = {
			is_overlord = yes
			AND = {
				is_subject = no
				has_variable = my_color
			}
		}
	}

	immediate = {
		#debug_log = start
		if = {
			limit = {
				is_overlord = yes
			}
			change_color_effect = { color = color }
		}
		if = {
			limit = {
				is_subject = no
				has_variable = my_color
			}
			change_color_effect = { color = var:my_color }
		}
		#debug_log = end
	}
}
