remove_party_leadership_effect = {
	if = {
		limit = {
			is_party_leader = yes
		}
		if = {
			limit = {
				employer = {
					is_republic = yes
					has_party = party:reactionary_bloc
				}
				character_party = reactionary_bloc
			}
			employer = {
				remove_party_leadership = reactionary_bloc
			}
		}
		else_if = {
			limit = {
				employer = {
					is_republic = yes
					has_party = party:conservative_bloc
				}
				character_party = conservative_bloc
			}
			employer = {
				remove_party_leadership = conservative_bloc
			}
		}
		else_if = {
			limit = {
				employer = {
					is_republic = yes
					has_party = party:radical_bloc
				}
				character_party = radical_bloc
			}
			employer = {
				remove_party_leadership = radical_bloc
			}
		}
		else_if = {
			limit = {
				employer = {
					is_republic = yes
					has_party = party:liberal_bloc
				}
				character_party = liberal_bloc
			}
			employer = {
				remove_party_leadership = liberal_bloc
			}
		}
		else_if = {
			limit = {
				employer = {
					is_republic = yes
					has_party = party:optimates_party
				}
				character_party = optimates_party
			}
			employer = {
				remove_party_leadership = party:optimates_party
			}
		}
		else = {
			add_gold = 1
		}
	}
}

randomise_starting_parties_effect = {
	every_country = {
		limit = {
			is_republic = yes
		}
		every_character = {
			switch_to_random_party = yes
			if = {
				limit = {
					is_party_leader = yes
				}
				remove_party_leadership_effect = yes
			}
		}
	}
}
